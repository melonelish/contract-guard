## runner.py 接口

```python
# backend/tests/eval/runner.py
"""
usage:
  python runner.py --mode mock            # mock 模式（默认）
  python runner.py --mode real --samples 3  # 真实模式，只抽 3 份
  python runner.py --mode real --all         # 全量（谨慎使用）
  python runner.py --compare run_001 run_002  # 对比两次结果
"""

import json, os, sys, time, argparse
from pathlib import Path

BASE_DIR = Path(__file__).parent
DATASETS_DIR = BASE_DIR / "datasets"
RESULTS_DIR = BASE_DIR / "results"
TEMPLATES_DIR = BASE_DIR / "templates"

# ── 样本优先级 ──
SAMPLE_PRIORITY = [
    # 高价值样本（优先抽测）
    "采购合同-设备采购",       # 最典型合同，风险场景完整
    "劳动合同-普通员工",       # 高频合同类型
    "技术开发合同-软件开发",     # 知识产权风险检测
    
    # 中价值样本
    "保密协议-双向保密",
    "租赁合同-办公场地",
    
    # 边界用例
    "空合同-仅签名页",          # 边界：几乎无条款
    "手写体扫描件",             # 边界：OCR 挑战
]

def run_sample(sample_path: Path, mode: str = "mock") -> dict:
    """运行单份样本评测——mock 模式返回模拟结果，real 模式调用真实接口"""
    if mode == "mock":
        return {
            "sample": sample_path.stem,
            "status": "passed" if "失败" not in sample_path.stem else "failed",
            "parser_clauses": 18, "analyzer_risks": 5, "total_duration_s": 14.2,
            "note": "mock result, not real LLM call"
        }
    # real 模式：抽样调用真实审查接口
    # 调用之前必须打印日志说明原因
    print(f"[REAL CALL] 调用真实审查: {sample_path.stem}")
    # ... 实际接口调用代码
    return {}

def batch_run(mode: str = "mock", max_samples: int = None, 
              priority_only: bool = False) -> list:
    """批量运行评测"""
    samples = sorted(DATASETS_DIR.glob("*.json"))
    if priority_only:
        priority_names = set(SAMPLE_PRIORITY)
        samples = [s for s in samples if s.stem in priority_names]
    if max_samples:
        samples = samples[:max_samples]
    
    results = []
    for s in samples:
        r = run_sample(s, mode)
        r["sample_file"] = s.name
        results.append(r)
    return results
```
